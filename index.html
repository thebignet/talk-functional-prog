public class Agios {

  private static int solde = 0;

  public static void main(String[] args) throws Exception {
    System.out.println("solde avant : " + solde + "€");
    int penalite = agios(2 * 5);
    solde = solde - penalite;
    System.out.println("solde après une pénalité de " + penalite + "€ : " + solde + "€");
    int penalite2 = agios(2 * 10);
    solde = solde - penalite2;
    System.out.println("solde après une pénalité de " + penalite2 + "€ : " + solde + "€");
  }

  private static int agios(int montant) {
    solde = -5;
    return montant;
  }
}
solde avant : 0€
solde après une pénalité de 10€ : -15€
solde après une pénalité de 20€ : -25€

public class Agios {
  public static void main(String[] args) throws Exception {
    Compte compte = new Compte(0)
        .agios(10)
        .agios(20);
    System.out.println("solde après deux agios : " + compte.solde + "€");
  }
  private static class Compte {
    final int solde;
    private Compte(int solde) {
      this.solde = solde;
    }
    Compte agios(int montant){
      return new Compte(solde-montant);
    }
  }
}

solde après deux agios : -30€
